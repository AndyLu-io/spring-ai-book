    // дҪҝз”Ё Spring AI + LangGraph жҰӮеҝөжһ„е»әе·ҘдҪңжөҒ // жіЁж„ҸпјҡSpring AI зӣ®еүҚйҖҡиҝҮ Advisor жңәеҲ¶ж”ҜжҢҒзұ»дјјеҠҹиғҪ // 1. е®ҡд№үе·ҘдҪңжөҒзҠ¶жҖҒ public class WorkflowState { private String userInput; private String analyzedIntent; private String processedData; private String finalOutput; private List<String> errors = new ArrayList<>(); private int retryCount = 0; // getters/setters... } // 2. е®ҡд№үиҠӮзӮ№пјҲжҜҸдёӘиҠӮзӮ№жҳҜдёҖдёӘеӨ„зҗҶжӯҘйӘӨпјү @Service public class OrderWorkflowNodes { private final ChatClient chatClient; // иҠӮзӮ№ 1пјҡж„ҸеӣҫеҲҶжһҗ public WorkflowState analyzeIntent(WorkflowState state) { String intent = chatClient.prompt() .system("еҲҶжһҗз”ЁжҲ·ж„ҸеӣҫпјҢиҝ”еӣһеҲҶзұ»пјҡжҹҘиҜў/дёӢеҚ•/жҠ•иҜү/е…¶д»–") .user(state.getUserInput()) .call().content(); state.setAnalyzedIntent(intent.trim()); return state; } // иҠӮзӮ№ 2пјҡи®ўеҚ•еӨ„зҗҶ public WorkflowState processOrder(WorkflowState state) { String result = chatClient.prompt() .system("ж №жҚ®з”ЁжҲ·ж„ҸеӣҫеӨ„зҗҶи®ўеҚ•") .user(state.getAnalyzedIntent() + ": " + state.getUserInput()) .call().content(); state.setProcessedData(result); return state; } // иҠӮзӮ№ 3пјҡз”ҹжҲҗеӣһеӨҚ public WorkflowState generateResponse(WorkflowState state) { String response = chatClient.prompt() .system("з”ЁеҸӢеҘҪзҡ„иҜӯж°”еӣһеӨҚз”ЁжҲ·") .user(state.getProcessedData()) .call().content(); state.setFinalOutput(response); return state; } } // 3. зј–жҺ’е·ҘдҪңжөҒ @Service public class OrderWorkflow { private final OrderWorkflowNodes nodes; public String execute(String userInput) { WorkflowState state = new WorkflowState(); state.setUserInput(userInput); // зәҝжҖ§е·ҘдҪңжөҒпјҡanalyzeIntent -> processOrder -> generateResponse state = nodes.analyzeIntent(state); state = nodes.processOrder(state); state = nodes.generateResponse(state); return state.getFinalOutput(); } }
