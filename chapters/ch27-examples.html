// ============================================ // 智能客服 REST API // ============================================ @RestController @RequestMapping("/api/v1/chat") public class ChatController { private final CustomerServiceBot bot; /** * 同步对话接口 * POST /api/v1/chat */ @PostMapping public ResponseEntity<ChatResponse> chat( @Valid @RequestBody ChatRequest request) { String reply = bot.chat(request.userId(), request.message()); return ResponseEntity.ok( new ChatResponse(request.userId(), reply, Instant.now())); } /** * 流式对话接口 - SSE * GET /api/v1/chat/stream */ @GetMapping(value = "/stream", produces = MediaType.TEXT_EVENT_STREAM_VALUE) public Flux<ServerSentEvent<String>> stream( @RequestParam String userId, @RequestParam String message) { return bot.streamChat(userId, message) .map(chunk -> ServerSentEvent.builder(chunk).build()); } } // 请求/响应 DTO public record ChatRequest( @NotBlank String userId, @NotBlank String message ) {} public record ChatResponse( String userId, String reply, Instant timestamp ) {}

Service 层（核心业务逻辑）

// ============================================ // 客服机器人核心 Service // 集成：意图分类 + RAG + 对话记忆 + 工单创建 // ============================================ @Service public class CustomerServiceBot { private final ChatClient chatClient; private final VectorStore knowledgeBase; private final ChatMemory chatMemory; private final TicketService ticketService; public CustomerServiceBot( ChatClient chatClient, VectorStore knowledgeBase, ChatMemory chatMemory, TicketService ticketService) { this.chatClient = chatClient; this.knowledgeBase = knowledgeBase; this.chatMemory = chatMemory; this.ticketService = ticketService; } /** * 主对话方法 * 流程：意图分类 → 路由到处理策略 → 返回结果 */ public String chat(String userId, String message) { // 1. 意图分类 String intent = classifyIntent(message); // 2. 根据意图路由到不同处理策略 return switch (intent) { case "FAQ" -> handleFaq(userId, message); case "COMPLAINT" -> handleComplaint(userId, message); case "ORDER" -> handleOrder(userId, message); default -> handleGeneral(userId, message); }; } /** * FAQ 处理 - 使用 RAG 从知识库检索答案 */ private String handleFaq(String userId, String message) { return chatClient.prompt() .system(""" 你是专业的客服助手。 基于知识库信息回答用户问题。 规则： 1. 只基于知识库中的信息回答，不要编造 2. 如果知识库没有相关信息，礼貌告知并建议联系人工客服 3. 回答要简洁、友好、专业 4. 适当使用 emoji 增加亲和力 """) // RAG: 从知识库检索相关文档 .advisors(new QuestionAnswerAdvisor(knowledgeBase, SearchRequest.query(message).withTopK(3))) // 对话记忆: 保留最近 10 条对话 .advisors(new MessageChatMemoryAdvisor( chatMemory, userId, 10)) .user(message) .call() .content(); } /** * 投诉处理 - 创建工单并安抚用户 */ private String handleComplaint(String userId, String message) { // 创建工单 Ticket ticket = ticketService.create(new Ticket( null, userId, "COMPLAINT", message, TicketStatus.OPEN, Instant.now())); return chatClient.prompt() .system(""" 你是客服助手，正在处理用户投诉。 工单已创建，编号：""" + ticket.id() + """ 请安抚用户情绪，告知工单已受理， 并说明预计处理时间。 """) .advisors(new MessageChatMemoryAdvisor( chatMemory, userId, 10)) .user(message) .call() .content(); } /** * 意图分类 - 用 LLM 做 zero-shot 分类 */ private String classifyIntent(String message) { return chatClient.prompt() .system(""" 将用户消息分类为以下意图之一： FAQ - 产品咨询、使用方法、功能说明 COMPLAINT - 投诉、不满、问题反馈 ORDER - 订单查询、物流、退换货 GENERAL - 其他通用对话 只返回意图标签，不要其他内容。 """) .user(message) .call() .content() .trim(); } /** * 流式对话 - 使用 Flux 实时返回 */ public Flux<String> streamChat(String userId, String message) { return chatClient.prompt() .system("你是专业的客服助手。") .advisors(new MessageChatMemoryAdvisor( chatMemory, userId, 10)) .user(message) .stream() .content(); } }

📈 扩展优化方向

  • 多轮引导：当用户描述模糊时，主动追问以获取更多信息
  • 情感分析：检测用户情绪，对愤怒用户优先转人工
  • 知识库更新：定时从 FAQ 文档自动同步到向量数据库
  • 效果评估：记录用户满意度评分，持续优化回答质量
  • 多租户：支持不同商家独立的知识库和对话记忆
⚠️ 项目难点与解决方案

难点 1：幻觉问题 -- AI 可能编造知识库中不存在的答案。解决方案：在 System Prompt 中严格要求"只基于知识库回答"，并在返回结果中标注信息来源。

难点 2：对话记忆过长 -- 多轮对话后 Token 消耗激增。解决方案：使用摘要压缩策略，定期将长对话压缩为关键信息摘要。

难点 3：意图分类不准 -- 用户表述模糊时分类错误。解决方案：使用 few-shot prompting 提供分类示例，或微调轻量级分类模型。

💣 踩坑复盘

坑1：Redis 对话记忆丢失 -- 上线后发现用户对话突然"失忆"。排查发现 Redis 默认 TTL 设置过短（30分钟），客服场景用户可能隔天继续咨询。解决：将 TTL 改为 7 天，并增加持久化存储兜底。

坑2：意图分类误判导致死循环 -- "我的订单一直没收到"被分类为 FAQ，反复从知识库检索却找不到答案。解决：增加"置信度阈值"机制，低于 0.7 的分类结果自动转人工。

// ============================================ // 文档智能分析 Service // 支持：上传解析 + 摘要生成 + 智能问答 + 结构化提取 // ============================================ @Service public class DocumentAnalysisService { private final ChatClient chatClient; private final VectorStore vectorStore; private final DocumentParser parser; public DocumentAnalysisService( ChatClient chatClient, VectorStore vectorStore, DocumentParser parser) { this.chatClient = chatClient; this.vectorStore = vectorStore; this.parser = parser; } /** * 上传并索引文档 * 解析 → 分块 → 向量化 → 存储 */ public DocIndexResult indexDocument(MultipartFile file) { // 1. 解析文档内容 List<Document> docs = parser.parse(file); // 2. 分块（每块 500 token，重叠 50 token） List<Document> chunks = docs.stream() .flatMap(d -> splitIntoChunks(d, 500, 50).stream()) .toList(); // 3. 存入向量数据库 vectorStore.accept(chunks); return new DocIndexResult( file.getOriginalFilename(), chunks.size(), Instant.now()); } /** * 文档摘要生成 */ public String summarize(String docId) { List<Document> docs = vectorStore .similaritySearch( SearchRequest.query("全文内容") .withFilterExpression("docId = '" + docId + "'") .withTopK(20)); String fullText = docs.stream() .map(Document::getContent) .collect(Collectors.joining("\n---\n")); return chatClient.prompt() .system(""" 你是文档分析专家。 请对以下文档生成结构化摘要： 1. 核心主题（一句话概括） 2. 关键要点（3-5 个要点） 3. 重要数据（如有） 4. 结论和建议 """) .user(fullText) .call() .content(); } /** * 基于文档的智能问答 */ public String askAboutDoc(String question) { return chatClient.prompt() .system(""" 你是文档分析助手。基于检索到的文档内容回答问题。 规则： 1. 引用具体的章节或段落 2. 如果文档中没有相关信息，明确说明 3. 用中文回答 """) .advisors(new QuestionAnswerAdvisor(vectorStore, SearchRequest.query(question).withTopK(5))) .user(question) .call() .content(); } /** * 结构化信息提取 - 使用 Structured Output */ public DocMetadata extractMetadata(MultipartFile file) { String content = parser.parseToText(file); return chatClient.prompt() .system(""" 从文档中提取以下元数据信息： - title: 文档标题 - author: 作者 - keywords: 关键词列表（最多5个） - summary: 一句话摘要 - pageCount: 页数 - category: 文档类别 """) .user(content) .call() .entity(DocMetadata.class); } } // 结构化输出 DTO public record DocMetadata( String title, String author, List<String> keywords, String summary, int pageCount, String category ) {} public record DocIndexResult( String filename, int chunkCount, Instant indexedAt ) {}

💣 踩坑复盘

坑1：PDF 解析乱码 -- 中文 PDF 用 Tika 解析后出现大量乱码和缺失段落。排查发现是 PDF 内嵌字体映射问题。解决：升级 Tika 版本并配置 OCR 后处理，对乱码段落用 OCR 重新识别。

坑2：分块过大导致检索不准 -- 初始设置每块 2000 token，结果检索出来的内容跨多个主题，LLM 生成答案时混淆信息。解决：改为按段落/章节语义分块，每块控制在 300-500 token，并增加重叠区域保持上下文连贯。

⚠️ 项目难点与解决方案

难点 1：SQL 注入风险 -- LLM 生成的 SQL 可能包含恶意操作。解决方案：(1) 只允许 SELECT；(2) 使用只读数据库账号；(3) 限制查询行数 LIMIT 100；(4) 设置查询超时 30 秒。

难点 2：复杂查询错误 -- LLM 可能生成错误的 JOIN 条件或聚合函数。解决方案：在 System Prompt 中提供详细的表结构和关系说明，使用 few-shot 示例引导生成正确的 SQL。

💣 踩坑复盘

坑1：SQL 注入绕过校验 -- 正则校验 `startsWith("SELECT")` 被绕过，用户输入"SELECT 1; DROP TABLE users--"。解决：增加多层防护 -- 白名单校验表名、禁止分号和注释符、使用 PreparedStatement 参数化查询。

坑2：复杂 JOIN 生成错误 -- LLM 生成的 SQL 在多表 JOIN 时经常出现笛卡尔积。解决：在 System Prompt 中明确表间关系（外键），并提供 3-5 个复杂查询的 few-shot 示例。

// ============================================ // 多语言翻译平台 // 流式翻译 + 术语管理 + 质量评估 // ============================================ @Service public class TranslationService { private final ChatClient chatClient; private final GlossaryRepository glossaryRepo; /** * 单文本翻译 - 带术语表 */ public TranslationResult translate( String text, String sourceLang, String targetLang) { // 1. 加载术语表 String glossary = glossaryRepo .findByLanguagePair(sourceLang, targetLang) .stream() .map(g -> g.source() + " → " + g.target()) .collect(Collectors.joining("\n")); // 2. 翻译 return chatClient.prompt() .system(""" 你是一位专业翻译员，精通多国语言。 翻译规则： 1. 将文本从 {source} 翻译成 {target} 2. 保持专业术语准确 3. 语句通顺自然，符合目标语言习惯 4. 专业术语对照表： {glossary} 输出 JSON： {"translation": "翻译结果", "confidence": 0.95} """.formatted( "source", sourceLang, "target", targetLang, "glossary", glossary)) .user(text) .call() .entity(TranslationResult.class); } /** * 流式翻译 - 实时返回翻译结果 */ public Flux<String> streamTranslate( String text, String source, String target) { return chatClient.prompt() .system("将以下文本从 {source} 翻译成 {target}。", Map.of("source", source, "target", target)) .user(text) .stream() .content(); } /** * 批量翻译 - 并行处理多段文本 */ public Flux<TranslationResult> batchTranslate( List<String> texts, String source, String target) { return Flux.fromIterable(texts) .parallel() .runOn(Schedulers.boundedElastic()) .map(text -> translate(text, source, target)) .sequential(); } /** * 文档翻译 - 保留 Markdown/HTML 格式 */ public String translateDocument( String content, String target) { return chatClient.prompt() .system(""" 翻译文档时严格遵守： 1. 保留原文的 Markdown/HTML 格式标签 2. 代码块不翻译 3. URL 和链接不翻译 4. 专业术语保持一致 5. 保持段落结构和缩进 """) .user("翻译为 {target}：\n{content}", Map.of( "target", target, "content", content)) .call() .content(); } } public record TranslationResult( String translation, double confidence ) {}

💣 踩坑复盘

坑1：Sentinel 与 Resilience4j 配置冲突 -- 同时使用两套限流框架时，Sentinel 的限流和 Resilience4j 的熔断互相干扰，导致正常请求被误拦。解决：明确职责划分 -- Sentinel 只负责网关层的全局 QPS 限流，Resilience4j 负责应用层的熔断降级和重试。

坑2：OpenTelemetry Span 泄漏 -- 高并发时 Jaeger 收到大量 Span 但部分 Span 未正确关闭，导致追踪链断裂。解决：使用 try-with-resources 模式管理 Span 生命周期，并配置 BatchSpanProcessor 的队列大小和导出超时。

坑3：灰度发布用户体验割裂 -- 同一用户在不同请求中命中不同版本，导致对话上下文不连贯。解决：使用一致性哈希，确保同一用户的请求始终路由到同一版本，并在 Redis 中缓存路由结果。

坑4：HPA 扩缩容抖动 -- CPU 指标波动导致 Pod 频繁扩缩，每次冷启动耗时 30 秒影响服务。解决：设置 HPA 的 stabilizationWindowSeconds=300，并增加自定义 QPS 指标作为扩缩依据。

🎓 面试高频题

👨‍💻 Q1: 如何设计一个生产级的 RAG 系统？

生产级 RAG 需要关注五个方面：(1) 文档解析 -- 使用 Apache Tika 等工具处理多种格式，智能分块（按语义而非固定长度）；(2) 向量检索 -- 选择合适的 Embedding 模型，混合检索（语义 + 关键词）提升召回率；(3) 上下文注入 -- 控制检索文档数量（TopK=3-5），避免上下文污染；(4) 答案生成 -- 严格要求基于检索内容回答，标注信息来源；(5) 质量评估 -- 使用 RAGAS 等框架评估忠实度和相关性。

👨‍💻 Q2: 如何处理 NL2SQL 的安全性问题？

多层防护策略：(1) 权限隔离 -- 使用只读数据库账号，禁止 DDL/DML 操作；(2) SQL 白名单 -- 只允许 SELECT 语句，使用正则校验；(3) 查询限制 -- 强制添加 LIMIT，设置查询超时；(4) 脱敏处理 -- 对返回结果中的敏感字段（手机号、身份证）自动脱敏；(5) 审计日志 -- 记录所有生成的 SQL 和查询结果，便于事后审查。

👨‍💻 Q3: 多 Agent 协作和单 Agent 有什么区别？什么时候用多 Agent？

单 Agent 适合任务明确、工具少于 10 个的场景，实现简单、延迟低。多 Agent 适合：(1) 任务涉及多个专业领域（如研究+写作+审核）；(2) 需要多视角分析（如代码审查从安全、性能、规范不同角度）；(3) 任务复杂度高，单 Agent 容易出错。多 Agent 的代价是更高的 Token 消耗和通信延迟，需要权衡利弊。建议先用单 Agent 实现 MVP，发现瓶颈后再拆分为多 Agent。

👨‍💻 Q4: 如何评估和优化 AI 应用的效果？

评估维度：(1) 准确性 -- 回答是否正确（人工评估 + 自动化测试）；(2) 延迟 -- 端到端响应时间（P50/P95/P99）；(3) 成本 -- Token 消耗和 API 调用费用；(4) 用户满意度 -- 收集用户评分和反馈。优化策略：Prompt 优化（最有效）、模型选择（简单任务用小模型）、缓存（相似查询复用结果）、流式输出（改善用户感知延迟）。
